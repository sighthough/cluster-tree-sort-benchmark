// Usage Example:
// const sortedData = optimizedClusterTreeSort([4.2, 1.5, 9.9, 0.3, 3.14]);

function optimizedClusterTreeSort(arr, maxBucketCapacity = 64) {
    if (arr.length <= 1) return arr;
    let min = arr[0], max = arr[0];
    for (let i = 1; i < arr.length; i++) {
        if (arr[i] < min) min = arr[i];
        if (arr[i] > max) max = arr[i];
    }
    let range = max - min;
    if (range === 0) return [...arr]; 

    let k = Math.ceil(Math.sqrt(arr.length));
    let bucketSize = range / k;
    let buckets = Array.from({ length: k }, () => []);

    for (let i = 0; i < arr.length; i++) {
        let num = arr[i];
        let bucketIdx = Math.floor((num - min) / bucketSize);
        if (bucketIdx >= k) bucketIdx = k - 1;
        buckets[bucketIdx].push(num);
    }

    let sortedResult = [];
    for (let i = 0; i < k; i++) {
        let bucket = buckets[i];
        if (bucket.length === 0) continue;
        if (bucket.length <= maxBucketCapacity) {
            bucket.sort((a, b) => a - b);
            for(let j = 0; j < bucket.length; j++) sortedResult.push(bucket[j]);
        } else {
            let subSorted = optimizedClusterTreeSort(bucket, maxBucketCapacity);
            for(let j = 0; j < subSorted.length; j++) sortedResult.push(subSorted[j]);
        }
    }
    return sortedResult;
}
